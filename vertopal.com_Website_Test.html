import pandas as pd
import plotly.express as px

# ---------- Load and Preprocess Data ----------
df = pd.read_csv("Compiled_Data.csv")

# Filter coordinates (Allamakee County bounding box)
df = df[
    (df["Latitude"] >= 43.1) & (df["Latitude"] <= 43.5) &
    (df["Longitude"] >= -91.7) & (df["Longitude"] <= -91.1)
].copy()

# Clean data
df = df.dropna(subset=["Latitude", "Longitude"])
df["Sex"] = df["Sex"].str.title()
df["Age"] = df["Age"].str.title()

# Interactive Map with Timelapse
fig = px.scatter_mapbox(
    df,
    lat="Latitude",
    lon="Longitude",
    color="Sex",
    animation_frame="Year",
    center={"lat": 43.3, "lon": -91.4},
    zoom=9,
    height=700,
    hover_data={"Sex": True, "Age": True, "Latitude": False, "Longitude": False, "Year": True}
)

fig.update_layout(
    mapbox_style="carto-positron",
    title="CWD Marker Map - Allamakee County, Iowa (Timelapse)"
)

# Export to standalone HTML (this is the file you upload to GitHub Pages)
fig.write_html("cwd_timelapse.html", full_html=True, include_plotlyjs='cdn')
